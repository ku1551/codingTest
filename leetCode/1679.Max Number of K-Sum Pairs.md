# 1679.Max Number of K-Sum Pairs

    class Solution {
        public int maxOperations(int[] nums, int k) {
            Map<Integer, Integer> cnt = new HashMap<>();
            int res = 0;
            for (int x : nums) {
                int need = k - x;
                if (cnt.getOrDefault(need, 0) > 0) {
                    cnt.put(need, cnt.get(need) - 1);
                    res++;
                } else {
                    cnt.put(x, cnt.getOrDefault(x, 0) + 1);
                }
            }
            return res;
        }
    }
