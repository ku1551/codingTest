## Code
    class Solution {
        public int climbStairs(int n) {
            if(n <= 3){
                return n;
            }       

            int prev2 = 2, prev1 = 3;
            int now = 0;
            for(int i=3; i<n;i++){
                now = prev1 + prev2;

                prev2 = prev1;
                prev1 = now;
            }
            return now;

        }
    }
