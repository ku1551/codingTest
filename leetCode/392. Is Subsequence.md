    class Solution:
        def isSubsequence(self, s: str, t: str) -> bool:

            idx = 0
            for i in range(len(t)):
                if idx < len(s) and s[idx] == t[i]:
                    idx += 1

            return idx == len(s)


## Java 
    class Solution {
        public boolean isSubsequence(String s, String t) {
            int n = s.length();
            int idx = 0;

            if (n == 0) return true;

            for(int i=0; i<t.length(); i++){
                if(s.charAt(idx) == t.charAt(i)){
                    idx++;
                    if(idx == n){
                        return true;
                    }
                }
            }
            return false;
        }
    }
